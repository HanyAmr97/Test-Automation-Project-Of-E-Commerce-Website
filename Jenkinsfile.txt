    agent any

    stages {

            steps {
                git 'https://github.com/HanyAmr97/Test-Automation-Project-Of-E-Commerce-Website.git'
            }

        stage('Build') {
            steps {

            }
        }

        stage('Test') {
            steps {
                echo 'mvn test -Dtest=SmokeTest'
            }
        }

        stage('Deploy') {
            steps {
                echo 'mvn deploy'
            }
        }
    }
}
